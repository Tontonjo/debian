# Commands to create a private and public key to allow passwordless SSH connexion between hosts

# Check if keys exist:
ls -al ~/.ssh/id_*.pub

# Generate keys
ssh-keygen -t rsa -b 4096 

# Copy ID to remote server
ssh-copy-id user@ipaddress
